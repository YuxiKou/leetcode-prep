public class Solution {
    public int Jump(int[] nums) {
        int jumps=0;
        int currentJumpEnd=0;
        int farthest=0;
            
        for(int x=0;x<nums.Length-1;x++)
        {
            
            if(farthest<nums[x]+x)
            {
                farthest=nums[x]+x;
            }
            
            //reached the end of the current jump
            if(x == currentJumpEnd)
            {
                jumps++;
                currentJumpEnd = farthest;
            }
        }
        
        return jumps;
        
    }
}
